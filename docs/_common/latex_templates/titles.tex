\RequirePackage{titlesec}
\RequirePackage{xcolor}

% ---------------------------------------------------------
% 定义 Neoway 深绿（适合 PDF 的正式商务色）
% ---------------------------------------------------------
\definecolor{nwyTitle}{RGB}{0,0,20}     % 主标题绿 (#2D6032)
\definecolor{nwyTitle2}{RGB}{0,0,20}   % 二级标题绿 (#336C38)
\definecolor{nwyTitle3}{RGB}{0,0,20}   % 三级标题绿 (#3D7A42)


% ---------------------------------------------------------
% Section 一级标题（3.1）
% ---------------------------------------------------------
\titleformat{\section}
  {\Large\bfseries\color{nwyTitle}}
  {\thesection}
  {0.8em}
  {}

\titlespacing*{\section}
  {0pt}{18pt plus 4pt minus 4pt}{10pt}

% ---------------------------------------------------------
% Subsection 二级标题（3.1.1）
% ---------------------------------------------------------
\titleformat{\subsection}
  {\bfseries\large\color{nwyTitle2}}
  {\thesubsection}
  {0.6em}
  {}

\titlespacing*{\subsection}
  {0pt}{14pt plus 3pt minus 3pt}{6pt}

% ---------------------------------------------------------
% Subsubsection 三级标题（3.1.1.1）
% ---------------------------------------------------------
\titleformat{\subsubsection}
  {\bfseries\normalsize\color{nwyTitle3}}
  {\thesubsubsection}
  {0.6em}
  {}

\titlespacing*{\subsubsection}
  {0pt}{10pt}{4pt}
