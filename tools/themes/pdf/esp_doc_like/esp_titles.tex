\RequirePackage{titlesec}
\RequirePackage{xcolor}

% ---------------------------------------------------------
% 颜色（根据 Neoway + ESP 风格统一
% ---------------------------------------------------------
\definecolor{espRed}{RGB}{176, 24, 32}     % 标题红色（ESP 同款）
\definecolor{espGray}{RGB}{120, 120, 120}  % 预留给 Chapter（但不使用）

% ---------------------------------------------------------
% ★ Section（一级标题：例如 3.1 …）
%   —— 红色编号 + 红色标题，无左竖线
% ---------------------------------------------------------
\titleformat{\section}
  {\Large\bfseries\color{espRed}}   % 风格：大一点、红色粗体
  {\thesection}                     % 显示编号“3.1”
  {0.8em}                           % 编号与标题的间距
  {}                                % 无竖线

\titlespacing*{\section}
  {0pt}{18pt plus 4pt minus 4pt}{10pt}

% ---------------------------------------------------------
% ★ Subsection（二级标题：例如 3.1.1 …）
% ---------------------------------------------------------
\titleformat{\subsection}
  {\bfseries\large\color{espRed}}
  {\thesubsection}
  {0.6em}
  {}

\titlespacing*{\subsection}
  {0pt}{14pt plus 3pt minus 3pt}{6pt}

% ---------------------------------------------------------
% ★ Subsubsection （三级标题：例如 3.1.1.1）
%   —— 建议也保持红色以维持一致性
% ---------------------------------------------------------
\titleformat{\subsubsection}
  {\bfseries\normalsize\color{espRed}}
  {\thesubsubsection}
  {0.6em}
  {}

\titlespacing*{\subsubsection}
  {0pt}{10pt}{4pt}
